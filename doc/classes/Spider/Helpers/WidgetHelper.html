        def prepare_scene(scene)
            scene = super
            if (self.is_a?(Widget))
                scene.widget = widget_to_scene(self)
            end
            scene.widgets = {}
            if (@widgets)
                @widgets.each do |id, w|
                    scene.widgets[id] = widget_to_scene(w)
                end
            end
            return scene
        end
