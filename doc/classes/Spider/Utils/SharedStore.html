        def self.get(type=nil, config=nil)
            type = Spider.conf.get('shared_store.type').to_sym unless type
            type = :memory unless type
            case type
            when :memory
                return MemorySharedStore.new(config)
            when :file
                return FileSharedStore.new(config)
            end
        end
