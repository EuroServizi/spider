        def inspect
            if (@action && @object)
                str = "#{@action} on #{@object} (#{object.class})\n"
                if (@dependencies.length > 0)
                    str += "-dependencies:\n"
                    @dependencies.each do |dep|
                        str += "---#{dep.action} on #{dep.object}\n"
                    end
                end
            else
                str = "Root Task"
            end
            return str
        end
