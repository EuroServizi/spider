        def self.get_storage(type, url)
            case type
            when 'db'
                adapter = url.match(/^(.+?):\/\//)[1]
                case adapter
                when 'sqlite'
                    storage = Db::SQLite.new(url)
                when 'oci8'
                    storage = Db::OCI8.new(url)
                when 'mysql'
                    storage = Db::Mysql.new(url)
                end
            end
        end
