        def self.included(model)
            model.element(:created_by, Spider::Security::User.model, :auto => true, :lazy => :accounting, :hidden => true)
            model.element(:modified_by, Spider::Security::User.model, :auto => true, :lazy => :accounting, :hidden => true)
            model.element(:created_date, DateTime, :auto => true, :lazy => :accounting, :hidden => true)

        def before_save(save_mode)
            super
            if (save_mode == :insert)
                set(:created_by, Spider::Security.user)
                set(:created_date, DateTime.now)
            end
            set(:modified_by, Spider::Security.user)
            set(:modified_date, DateTime.now)
        end
