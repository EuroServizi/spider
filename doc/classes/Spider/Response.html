        def prepare_headers
            @headers['Set-Cookie'] ||= []
            @cookies.each do |k, v|
                h = "#{k}=#{v}"
                h += '; expires='+v.expires.strftime("%a, %b %d %Y %H:%M:%S %Z") if (v.expires.respond_to?(:strftime))
                h += "; path=#{v.path}" if (v.path)
                h += "; domain=#{v.domain}" if (v.domain)
                h += "; secure" if (v.secure)
                @headers['Set-Cookie'] << h
            end
            Spider::Logger.debug("HEADERS:")
            Spider::Logger.debug(@headers)
        end
