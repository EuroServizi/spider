        def self.close_resources
            return unless @resources
            @mutex ||= Mutex.new
            @mutex.synchronize do
                @resources.each do |res|
                    @resource.close if resource && resource.respond_to?(:close) && !resource.closed?
                end
                @resources.clear
            end
        end
