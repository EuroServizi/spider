        def body(&proc)
            b = @body.is_a?(String) ? StringIO.new(@body) : @body
            return nil unless b
            while (buf = b.read(BUFSIZE))
                yield buf
            end
        end

        def read_body
            return @body if @body.is_a?(String)
            b = ''
            self.body do |buf|
                b += buf
            end
            @body = b
        end
