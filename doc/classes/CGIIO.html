    def send_headers
        Spider::Logger.debug("CGI sending headers:")
        @controller_response.prepare_headers
        @headers_sent = true
        @out << "Status: #{@controller_response.status}\n"
        @controller_response.headers.each do |key, val|
            @out << "#{key}: #{val}\n"
        end
        @out << "\n"
    end
